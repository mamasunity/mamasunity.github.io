Jupyter notebookで、二値エントロピー関数をグラフ化する

import numpy as np
import matplotlib.pylab as plt

p = np.linspace(0.001, 0.999)

plt.plot(p, -p * np.log2(p) - (1-p) * np.log2(1 - p))

plt.xlabel("Binary entropy function")
plt.ylabel('-plogp-(1-p)log(1-p)')
plt.grid()

plt.show()
